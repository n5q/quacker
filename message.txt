```sql
CREATE TABLE Parcels (
track_number        INTEGER NOT NULL,
delivery_address    TEXT,
drop_time           DATETIME,
goods               TEXT,
weight              REAL,
PRIMARY KEY (track_number)
);

CREATE TABLE Facilities (
id                  INTEGER NOT NULL,
city                TEXT,
address             TEXT,
manager_id          INTEGER NOT NULL,
FOREIGN KEY (manager_id) REFERENCES Employees(id),
PRIMARY KEY (id)
);

CREATE TABLE Employees (
id                  INTEGER NOT NULL,
name                TEXT,
address             TEXT,
role                TEXT,
works_at            INTEGER NOT NULL,
FOREIGN KEY (works_at) REFERENCES Facilities(id),
PRIMARY KEY (id)
);

CREATE TABLE Attempt (
delivery_time       DATETIME NOT NULL,
driver              TEXT,
status              TEXT,
feedback            TEXT,
track_number        INTEGER NOT NULL,
FOREIGN KEY (track_number) REFERENCES Parcels(track_number)
ON UPDATE CASCADE
ON DELETE CASCADE,
PRIMARY KEY (delivery_time, track_number)
);

CREATE TABLE Enter (
entry_time          DATETIME,
next                TEXT,
track_number        INTEGER NOT NULL,
facility_id         INTEGER NOT NULL,
FOREIGN KEY (track_number) REFERENCES Parcels(track_number),
FOREIGN KEY (facility_id) REFERENCES facilities(id),
PRIMARY KEY (track_number, facility_id)
);

CREATE TABLE Exit (
exit_time           DATETIME,
transportation      TEXT,
track_number        INTEGER NOT NULL,
facility_id         INTEGER NOT NULL,
FOREIGN KEY (track_number) REFERENCES Parcels(track_number),
FOREIGN KEY (facility_id) REFERENCES facilities(id),
PRIMARY KEY (track_number, facility_id)
);

CREATE TABLE Regular (
track_number        INTEGER NOT NULL,
FOREIGN KEY (track_number) REFERENCES Parcels(track_number)
ON UPDATE CASCADE
ON DELETE CASCADE,
PRIMARY KEY (track_number)
);

CREATE TABLE Expedited (
track_number        INTEGER NOT NULL,
date                DATE,
country             TEXT,
total_value         REAL,
FOREIGN KEY (track_number) REFERENCES Parcels(track_number)
ON UPDATE CASCADE
ON DELETE CASCADE,
PRIMARY KEY (track_number)
);

CREATE TABLE Supervise (
supervising         INTEGER NOT NULL,
supervisor          INTEGER NOT NULL,
FOREIGN KEY (supervisor) REFERENCES Employees(id),
FOREIGN KEY (supervising) REFERENCES Employees(id),
PRIMARY KEY (supervising, supervisor)
);
```
